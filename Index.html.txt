  <script>
    // ---------------------------------------------------------
    // PASTE YOUR GOOGLE DEPLOYMENT URL HERE INSIDE THE QUOTES:
    const API_URL = "Paste_Your_Google_Script_URL_Here"; 
    // ---------------------------------------------------------

    function searchResult() {
      const seatRaw = document.getElementById('seatInput').value;
      const seat = seatRaw ? seatRaw.trim() : "";
      const btn = document.querySelector('.btn-search');

      if (!seat) {
        document.getElementById('seatInput').focus();
        return;
      }

      document.getElementById('loader').style.display = 'block';
      document.getElementById('resultArea').style.display = 'none';
      document.getElementById('errorMsg').style.display = 'none';
      btn.disabled = true;
      btn.innerHTML = 'Searching...';

      fetch(API_URL + "?seat=" + encodeURIComponent(seat))
        .then(response => response.json())
        .then(data => {
          btn.disabled = false;
          btn.innerHTML = 'Search Result';
          document.getElementById('loader').style.display = 'none';

          if (data.found) {
            document.getElementById('resIdentifier').innerText = "Seat No: " + data.seat; 
            document.getElementById('resRank').innerText = "#" + data.rank;
            document.getElementById('resTotal').innerText = data.total;
            document.getElementById('resPercent').innerText = data.percent;
            document.getElementById('resEval').innerText = data.evaluation; 

            const grade = data.grade ? data.grade.toString() : "-";
            const firstChar = grade.charAt(0).toUpperCase();
            let gradeClass = 'bg-F'; 

            if (firstChar === "A") gradeClass = "bg-A";
            else if (firstChar === "B") gradeClass = "bg-B";
            else if (firstChar === "C") gradeClass = "bg-C";
            else if (firstChar === "D") gradeClass = "bg-D";

            const circle = document.getElementById('gradeCircle');
            circle.innerText = grade;
            circle.className = 'grade-circle ' + gradeClass;

            document.getElementById('resultArea').style.display = 'block';
          } else {
            document.getElementById('errorMsg').style.display = 'block';
          }
        })
        .catch(error => {
          console.error('Error:', error);
          btn.disabled = false;
          btn.innerHTML = 'Search Result';
          document.getElementById('loader').style.display = 'none';
          alert("Error connecting. Verify your Google Script URL.");
        });
    }

    document.getElementById("seatInput").addEventListener("keypress", function(event) {
      if (event.key === "Enter") searchResult();
    });
  </script>
